\recette{Aeropress Kahvi - James Hoffman}
\begin{center}
     \preptime{1 min} \cooktime{3 min} \people{1}
\end{center}

\recipebody{Maukas aeropress kahvi James Hoffmanin tapaan.}

\recipe{
  Vasta jauhettua kahvia & \unit[11]{g}\\
  99 asteista vettä & \unit[200]{g}\\
}{
    \item Jauha haluamasi kahvipavut keskihienoksi
    \item Laita aeropressiin hautumaan 2 minuutiksi
    \item Kevyesti pyöritä pressiä
    \item Anna hautua vielä 30 sekuntia
    \item Paina pressi kevyesti pohjaan. Noin 30 sekuntia.

}

\info{Kahvi ja vesi kannattaa mitata keittövaa'alla, jotta kahvi on joka kerralla samanlaista. Veden lämpötilaa kannattaa hieman laskea (noin 5 astetta), mikäli käytössä on tumma paahtoinen kahvi. Aeropressillä tehty kahvi muistuttaa maultaan hieman espressoa, joten valmista juomaa voi jatkaa hieman kuumalla vedellä, jolloin se vastaa enemmän perinteistä suodatinkahvia.}
