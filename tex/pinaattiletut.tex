\recipe[Nam nam vihreitä lettuja]{Pinaattiletut}[Letut kannattaa tarjoilla puollukoiden ja raejuuston kanssa]
\serves{4}
\preptime{5 min}
\cooktime{30 min}

\begin{step}
    3~kpl Munia
    5dl Maito
    3dl Vehnäjauhoja
    1~tl Suolaa
    150g Pinaattia
    1~rkl Öljyä
    \method
    Silppua pinaatti

    Riko munat kulhoon

    Sekoita ainekset keskennän

    Anna turvota noin 30 minuuttia

    Paista letut
\end{step}
